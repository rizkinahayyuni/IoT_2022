#include <Arduino.h>
#include <SimpleDHT.h>

#define pinDHT 1 

byte temperature = 0;
byte humidity = 0;

SimpleDHT11 dht11(D1); 

void KelembabanSuhu()
{
  int err = SimpleDHTErrSuccess;

  if ((err = dht11.read(&temperature, &humidity, NULL)) != SimpleDHTErrSuccess)
  {
    Serial.print("Pembacaan DHT11 gagal, err=");
    Serial.println(err);
    delay(1000);
    return;
  }

  Serial.print("Temperatur: ");
  Serial.print((int)temperature);
  Serial.print(" 째C | ");
  Serial.print((1.8 * (double)temperature) + 32);
  Serial.print(" 째F | ");
  Serial.print((0.8 * (double)temperature));
  Serial.print(" 째R | ");
  Serial.print((double)temperature + 273.15);
  Serial.println(" 째K");
  Serial.print("Kelembapan : ");
  Serial.print((int)humidity);
  Serial.println(" % H");
  Serial.println("================================================");

  delay(1500);
}

void setup()
{
  Serial.begin(115200);
  Serial.println("Simple DHT and Convert");
  delay(1000);
}

void loop()
{
  KelembabanSuhu();
}
